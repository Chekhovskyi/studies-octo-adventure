# coding=utf-8
from numpy import *

def left_endpoint(x, y, f, a, b):
    return ((b-a)/len(x)) * sum(y[0:len(y)-1])

def right_endpoint(x, y, f, a, b):
    return ((b-a)/len(x)) * sum(y[1:])


def trapezoidal(x, y, f):
    h = x[1]-x[0]
    doubles = [v * 2 for v in y[1:len(y)-1]]
    return (h/2) * (y[0] + sum(doubles) + y[-1])

def simpson(x, y, f):
    h = x[1]-x[0]
    s1 = sum(y[1:-1:2])
    s2 = sum(y[2::2])
    return (h/3) * (y[0] + y[-1] + 4*s1 + 2*s2)

def f_28(x):
    return exp(x) * sqrt(x)
a = 5
b = 10
f = f_28 

x_etalone = arange(a, b + 0.00001, 0.00001)
y_etalone = [f(v) for v in x_etalone]
tr_etalone = trapz(y=y_etalone, x=x_etalone)

segments = [10, 100, 1000]
xs = [linspace(a, b, s) for s in segments]
ys = [[f(v) for v in x] for x in xs]

les = [left_endpoint(x, ys[i], f, a, b) for (i, x) in enumerate(xs)]
res = [right_endpoint(x, ys[i], f, a, b) for (i, x) in enumerate(xs)]
trs = [trapezoidal(x, ys[i], f) for (i, x) in enumerate(xs)]
sis = [simpson(x, ys[i], f) for (i, x) in enumerate(xs)]

display_table(["", "10", "100", "1000"],[
